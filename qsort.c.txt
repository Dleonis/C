#include<stdio.h>
#include<stdlib.h>
#include<string.h>
int compare1(const void *a, const void *b)//ÅÐ¶Ïint 
{
	return (*(int *)a - *(int *)b);
}
int compare2(const void *a, const void *b)//ÅÐ¶Ïchar
{
	return (*(char *)a - *(char *)b);
}
int compare3(const void *a, const void *b)//ÅÐ¶Ïfloat
{
	return (*(float *)a > *(float *)b ? 1 : -1);
}

void show(const int *a, const int len)
{
	int i = 0;
	for (i = 0; i < len; i++)
	{
		printf("%d ", a[i]);
	}
	printf("\n");
}

void show1(const float *a, const int len)
{
	int i = 0;
	for (i = 0; i < len; i++)
	{
		printf("%f ", a[i]);
	}
	printf("\n");
}

int main()
{
	int arr1[] = { 10, 3, 5, 9, 6, 8, 11, 20 };
	char arr2[] = "bdcafeg";
	float arr3[] = { 2.1, 4.3, 1.2, 3.1, 5.4, 7.5, 6.8, 3.5 };
	int len1 = sizeof(arr1) / sizeof(arr1[0]);
	int len2 = strlen(arr2);
	int len3 = sizeof(arr3) / sizeof(arr3[0]);

	int(*comp1)(const void *a, const void *b);
	show(arr1, len1);
	comp1 = &compare1;
	qsort(arr1, len1, sizeof(arr1[0]), comp1);
	show(arr1, len1);

	int(*comp2)(const void *a, const void *b);
	printf("%s\n", arr2);
	comp2 = &compare2;
	qsort(arr2, len2, sizeof(arr2[0]), comp2);
	printf("%s\n", arr2);

	int(*comp3)(const void *a, const void *b);
	comp3 = &compare3;
	show1(arr3, len3);
	qsort(arr3, len3, sizeof(arr3[0]), comp3);
	show1(arr3, len3);
	system("pause");
	return 0;
}
